        <!-- carousel control bootstrap -->
        <script>
            let ci = 0;
            let autoSlideInterval;

            // Function to show the current slide
            function showSlide(index) {
                const slides = document.querySelectorAll('.carousel-item');
                const totalSlides = slides.length;

                // Prevent index overflow
                if (index >= totalSlides) {
                    ci = 0;
                } else if (index < 0) {
                    ci = totalSlides - 1;
                } else {
                    ci = index;
                }

                const carouselItems = document.querySelector('.carousel-items');
                carouselItems.style.transform = `translateX(-${ci * 100}%)`;
            }

            // Function to navigate to the previous slide
            function prevSlide() {
                showSlide(ci - 1);
                resetAutoSlide(10000); // Reset auto-slide timer to 10 seconds
            }

            // Function to navigate to the next slide
            function nextSlide() {
                showSlide(ci + 1);
                resetAutoSlide(10000); // Reset auto-slide timer to 10 seconds
            }

            // Function to reset the auto-slide timer
            function resetAutoSlide(delay = 5000) {
                clearInterval(autoSlideInterval); // Clear the current interval
                autoSlideInterval = setInterval(nextSlide, delay); // Set a new interval with the given delay
            }

            // Initial load
            showSlide(ci);

            // Start the auto-slide timer (default interval of 5 seconds)
            autoSlideInterval = setInterval(nextSlide, 3000);
        </script>
